#include <cs50.h>
#include <stdio.h>

int main(void)
{
    int n; 
    do
    {
        n = get_int("Enter the size: ");
    }
    while (n < 1);
    for (int i = 0; i < n; i++)
    {
        for (int k = 0; k < n - i; k++)
        {
            printf(" ");
        }
        for (int j = 0; j <= i; j++)
        {
            printf("#");
        }
        
        printf("\n");
    }

}
